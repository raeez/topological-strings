\section{Factorization Current Conventions for the CE/PV Interface}
\label{sec:app-factorization-current-conventions}

This appendix fixes the algebraic source/target interface carried by
the factorization-current theorem.  The interface is used before the
admissible bar-cobar/enveloping criterion: interval Hamiltonian
currents form the CE source, distributional Matlis principal parts form
the cotangent current target, and the CE/PV generator dictionary is
\[
  c_\lambda\longmapsto\theta_\lambda,\qquad
  u_{a(t)dt\otimes f}\longmapsto B_f(a).
\]
The proved content is the reduced algebraic current model after
de Rham-current contraction.  The unreduced BV/QME current lift and the
Costello brane-defect graph theorem are separate obstruction problems,
recorded in Remark~\ref{rmk:unreduced-lift-status},
Appendix~\ref{sec:app-unreduced-bv-qme}, and
Problem~\ref{prob:analytic-graph-realization}.
All completed symmetric algebras in this appendix are interpreted
degreewise in the chosen current category; finite symmetric powers are
formed before completion, and brackets extend continuously from the
displayed length-one current formulas.

\begin{defn}[Interval Hamiltonian current algebra]
\label{def:app-factorization-current-source}
  For an open interval \(I\subset\R\), set
  \[
    \mathfrak h_I
      =\Omega^\bullet_c(I)\widehat\otimes\mathfrak h ,
  \]
  with differential \(d_I=d_{\mathrm{dR}}\otimes1\).  The Hamiltonian
  Lie algebra \(\mathfrak h\) is concentrated in degree \(0\), so the
  graded bracket is
  \[
    [\alpha\otimes f,\beta\otimes g]
      =\alpha\wedge\beta\otimes\{f,g\},
      \qquad
      \alpha,\beta\in\Omega^\bullet_c(I),\ f,g\in\mathfrak h .
  \]
  For degree-zero test functions \(a,b\in\Omega^0_c(I)\),
  \[
    [a\otimes f,b\otimes g]=ab\otimes\{f,g\}.
  \]
  The cosheaf structure for inclusions of intervals is extension by
  zero on \(\Omega^\bullet_c\), tensored with the identity on
  \(\mathfrak h\).  Polynomial formulas use the dense scalar-reduced
  subalgebra \(\bar A=\C[z_1,z_2]/\C\cdot1\subset\mathfrak h\).
  For \(j:I\hookrightarrow J\), extension by zero is smooth because
  every element of \(\Omega_c^\bullet(I)\) has support compactly
  contained in \(I\).  On CE cochains the induced map is the dual
  coproduct/pullback; the covariant factorization structure belongs to
  the compact-support current target, not to a restriction map on raw
  CE cochains.

  The notation \(a(t)dt\otimes f\) is the oriented density label paired
  with distributional cotangent currents.  It does not convert
  \(a\in\Omega^0_c(I)\) into a degree-one Hamiltonian generator of
  \(\mathfrak h_I\).
\end{defn}

\begin{defn}[Distributional Matlis current dual]
\label{def:app-factorization-current-dual}
  Let
  \[
    \mc P=\operatorname{Ann}_{\mathrm{res}}(\C\cdot1)
      \cong\mathfrak h^\vee_{\mathrm{cont}}
  \]
  be the reduced Matlis principal-part module of
  Section~\ref{sec:app-matlis-principal-parts}.  Its interval current
  form is
  \[
    \mathcal P_I=\mathcal D'_c(I)\widehat\otimes\mc P .
  \]
  This is the selected support-local current dual used after
  contraction, not an assertion that the full strong dual of
  \(\Omega_c^\bullet(I)\widehat\otimes\mathfrak h\) has been imported.
  The pairing with degree-zero Hamiltonian currents is
  \[
    \langle B\otimes\rho,\;a\otimes f\rangle
      =B(a)\operatorname{Res}(\rho f),
      \qquad
      B\in\mathcal D'_c(I),\ \rho\in\mc P.
  \]
  If \(B=b(t)dt\) is a regular compactly supported density, then
  \[
    B(a)=\int_Ia(t)b(t)\,dt .
  \]

  The shifted cotangent interval Lie algebra is
  \[
    \mathfrak g_I=\mathfrak h_I\ltimes\mathcal P_I[1],
  \]
  where \(\mathfrak h_I\) acts on \(\mathcal P_I\) by the product of
  the \(C_c^\infty(I)\)-module operation \(B\mapsto aB\) on compactly
  supported currents and the Matlis coadjoint action
  on \(\mc P\).  On degree-zero labels this action is
  \[
    (a\otimes f)\cdot(B\otimes\rho)
      =(aB)\otimes(f\cdot\rho).
  \]
\end{defn}

\begin{lem}[Multiplication of a current by a test function]
\label{lem:app-factorization-current-product}
  For \(a\in C^\infty_c(I)\) and \(B\in\mathcal D'_c(I)\), define
  \[
    (aB)(\varphi)=B(a\varphi),
    \qquad \varphi\in C^\infty_c(I).
  \]
  Then \(aB\in\mathcal D'_c(I)\),
  \[
    \operatorname{supp}(aB)\subset
    \operatorname{supp}(a)\cap\operatorname{supp}(B),
  \]
  and \(a(bB)=(ab)B\).  If \(B=b(t)dt\) is regular, then
  \(aB=(ab)(t)dt\).  No product of two distributions occurs in the
  reduced current model.
\end{lem}

\begin{proof}
  The function \(a\varphi\) is a compactly supported smooth test
  function, so precomposition by \(\varphi\mapsto a\varphi\) defines a
  compactly supported distribution.  If \(\varphi\) is supported away
  from \(\operatorname{supp}(a)\cap\operatorname{supp}(B)\), then
  \(a\varphi\) is supported away from \(\operatorname{supp}(B)\), hence
  \(B(a\varphi)=0\).  Associativity follows from
  \(a(b\varphi)=(ab)\varphi\).  The regular-density formula is the
  displayed integral identity.
\end{proof}

\begin{defn}[Current CE/PV coordinates]
\label{def:app-factorization-current-coordinates}
  Let \(\lambda=B\otimes\rho\in\mathcal P_I\).  The Hamiltonian CE
  coordinate dual to the Hamiltonian current through \(\lambda\) is
  written
  \[
    c_\lambda,\qquad |c_\lambda|=1 .
  \]
  Its PV/current target coordinate is
  \[
    \theta_\lambda=\theta_{B\otimes\rho}:=\Theta_\rho(B).
  \]

  The label \(a(t)dt\otimes f\) denotes the functional on
  \(\mathcal P_I\)
  \[
    B\otimes\rho\longmapsto B(a)\operatorname{Res}(\rho f).
  \]
  The CE coordinate dual to the shifted cotangent generator through this
  functional is
  \[
    u_{a(t)dt\otimes f},\qquad |u_{a(t)dt\otimes f}|=0.
  \]
  Its PV/current target coordinate is the smeared Hamiltonian boundary
  observable
  \[
    O_{a,f}=B_f(a).
  \]
  In the brane realization,
  \[
    B_f(a)=
      \int_I a(t)\,
      \overline{\operatorname{Tr}f(\phi_1(t),\phi_2(t))}\,dt .
  \]
\end{defn}

\begin{defn}[Enlarged reduced defect-current target]
\label{def:app-factorization-enlarged-current-target}
  The reduced defect-current target attached to \(I\) is
  \[
    A_\partial^{\mathrm{cur}}(I)
      =
      \widehat{\mathbf S}(\mathfrak h_I)
      \widehat\otimes
      \widehat{\mathbf S}(\mathcal P_I[1]).
  \]
  This is the same reduced principal-part target denoted
  \(A_\partial^{\mathrm{pp}}(I)\) in the main text; the superscript
  \(\mathrm{cur}\) records the current-interface role in this appendix.
  It enlarges the Hamiltonian target
  \(A_\partial^{\mathrm{Ham}}(I)=\widehat{\mathbf S}(\mathfrak h_I)\)
  by adjoining the odd principal-part current generators
  \(\theta_\lambda=\Theta_\rho(B)\).  The structure maps for
  inclusions of intervals are extension by zero on
  \(\Omega^\bullet_c(I)\) and on \(\mathcal D'_c(I)\).  The
  factorization product on disjoint intervals is the completed
  graded-commutative product after extension by zero.
\end{defn}

\begin{thm}[Universal algebraic current interface]
\label{thm:app-factorization-universal-current-interface}
  For each interval \(I\), the generator assignment
  \[
    \Phi_I^{\mathrm{cur}}(c_\lambda)=\theta_\lambda,\qquad
    \Phi_I^{\mathrm{cur}}(u_{a(t)dt\otimes f})=B_f(a)
  \]
  extends uniquely to a continuous morphism of completed
  graded-commutative algebras from the interval CE cofactorization source
  \(C^\bullet_{\mathrm{CE},\mathrm{cont}}(\mathfrak g_I)\) to the
  reduced defect-current central-operation target
  \(A_\partial^{\mathrm{cur}}(I)\).  On length-one current generators
  the target open-current \(P_0\) brackets are
  \[
  \begin{aligned}
    \{B_f(a),B_g(b)\}
      &=B_{\{f,g\}}(ab),\\
    \{B_f(a),\theta_{B\otimes\rho}\}
      &=\theta_{aB\otimes(f\cdot\rho)}
        =\Theta_{f\cdot\rho}(aB),\\
    \{\theta_{B\otimes\rho},\theta_{C\otimes\sigma}\}
      &=0 .
  \end{aligned}
  \]
  The factorization product and the reduced current source-target
  \(P_0\)-equivariance defects in this reduced current target are strict:
  \[
    H_{\mathrm{fact}}=0,\qquad H_{P_0}=0.
  \]
  This is not a zero Poisson-centre statement for overlapping supports;
  the nonzero Hamiltonian current bracket is the moment-map bracket
  displayed above.  Thus \(u_{a(t)dt\otimes f}\mapsto B_f(a)\) and
  \(c_\lambda\mapsto\theta_\lambda\) form the current-level
  source/target interface used before the admissible bar-cobar/enveloping
  criterion.
\end{thm}

\begin{proof}
  The CE algebra of
  \(\mathfrak g_I=\mathfrak h_I\ltimes\mathcal P_I[1]\) is the
  completed symmetric algebra on the continuous dual shifted by
  \([-1]\).  The coordinate \(c_\lambda\) has degree \(1\), and the
  coordinate \(u_{a(t)dt\otimes f}\) has degree \(0\).  The target
  \(A_\partial^{\mathrm{cur}}(I)\) is the completed symmetric algebra on
  the corresponding current coordinates \(\theta_\lambda\) and
  \(B_f(a)\).  Hence the displayed assignment extends uniquely and
  continuously to the completed graded-commutative algebras.

  The CE differential and the target current brackets use the same
  semidirect Lie structure constants.  The Hamiltonian-current bracket
  in Definition~\ref{def:app-factorization-current-source} gives
  \[
    [a\otimes f,b\otimes g]=ab\otimes\{f,g\},
  \]
  and the boundary realization gives the first displayed bracket.  The
  action of \(a\otimes f\) on \(B\otimes\rho\) is
  \((aB)\otimes(f\cdot\rho)\), giving the second displayed bracket by
  negative transpose under the product pairing
  \[
    \langle B\otimes\rho,\;a\otimes f\rangle
      =B(a)\operatorname{Res}(\rho f).
  \]
  The shifted cotangent current ideal is abelian, giving the third
  bracket.  These three identities are exactly the current form of the
  CE/PV generator identities \(c_\lambda\mapsto\theta_\lambda\) and
  \(u\mapsto O\).  Since the CE differential is a derivation and the
  target bracket is a biderivation, the generator identities determine
  the full completed current interface.

  Factorization compatibility is strict in the cofactorization-source
  convention.  Extension by zero on current Lie algebras induces the CE
  coproduct, and the target product is compatible with pointwise
  products \(ab\) and with the \(C_c^\infty(I)\)-module operation
  \(B\mapsto aB\).
  If two intervals are disjoint, the products have empty support and
  the mixed brackets vanish.  The target product is graded
  commutative, so multiplication by an image generator has zero
  Hochschild product-centrality defect.

  For reduced current \(P_0\)-equivariance, let \(x\) be a length-one source generator and
  let \(Y\) be a homogeneous monomial in the CE source.  The defect
  \[
    D_x(Y)=
    \{\Phi_I^{\mathrm{cur}}(x),\Phi_I^{\mathrm{cur}}(Y)\}
      -\Phi_I^{\mathrm{cur}}(\operatorname{ad}_xY)
  \]
  is a derivation in \(Y\).  The three generator bracket identities
  give \(D_x(Y)=0\) for every length-one \(Y\).  The Leibniz rule gives
  \(D_x(Y)=0\) for every monomial, and continuity extends this to the
  completed algebra.  Thus \(H_{\mathrm{fact}}\) and \(H_{P_0}\) are
  zero equivariance-defect homotopies in the reduced current model.
\end{proof}

\begin{cor}[Principal-part current brackets in monomial coordinates]
\label{cor:app-factorization-principal-part-current-brackets}
\label{prop:app-factorization-principal-part-bracket}
  For \(f=z_1^p z_2^q\), \(a\in C^\infty_c(I)\),
  \(B\in\mathcal D'_c(I)\), and
  \(\rho_{i,j}=z_1^{-i-1}z_2^{-j-1}dz_1dz_2\), one has
  \[
    \{B_{z_1^p z_2^q}(a),\Theta_{\rho_{i,j}}(B)\}
      =
      -(pj-qi+p-q)
      \Theta_{\rho_{i-p+1,j-q+1}}(aB),
  \]
  with negative-index targets equal to zero.  If \(B=b(t)dt\) is
  regular, then the interval factor is \((ab)(t)dt\).
\end{cor}

\begin{proof}
  Proposition~\ref{prop:app-matlis-coadjoint-action} gives
  \[
    z_1^p z_2^q\cdot\rho_{i,j}
      =-(pj-qi+p-q)\rho_{i-p+1,j-q+1},
  \]
  with negative-index terms projected to zero in the principal-part
  module.  Substituting this action into the second bracket of
  Theorem~\ref{thm:app-factorization-universal-current-interface} gives
  the displayed formula.  The regular-current statement is
  Lemma~\ref{lem:app-factorization-current-product}.
\end{proof}

\begin{cor}[Smeared Hamiltonian source convention]
\label{cor:app-factorization-source-of-B}
\label{prop:app-factorization-source-of-B}
  The smeared boundary Hamiltonian observable is sourced by the
  shifted-cotangent CE coordinate:
  \[
    u_{a(t)dt\otimes f}\longmapsto
    B_f(a)
      =\int_I a(t)\,
        \overline{\operatorname{Tr}
        f(\phi_1(t),\phi_2(t))}\,dt .
  \]
  The Hamiltonian CE coordinate \(c_\lambda\) maps to the PV/current
  vector \(\theta_\lambda\), not to \(B_f(a)\).  Consequently
  \[
    \{B_f(a),B_g(b)\}=B_{\{f,g\}}(ab)
  \]
  is the degree-zero moment-map shadow of the CE/PV identity
  \(c_\lambda\mapsto\theta_\lambda,\ u\mapsto O\).
\end{cor}

\begin{proof}
  The coordinate \(u_{a(t)dt\otimes f}\) is dual to the shifted
  cotangent generator through the pairing with
  \(\mathcal P_I\).  Under the CE/PV isomorphism it maps to the
  function coordinate \(O_{a,f}\), realized on the brane by the
  displayed trace integral.  The coordinate \(c_\lambda\) is dual to a
  Hamiltonian current and maps to the vector/current coordinate
  \(\theta_\lambda\).  The bracket identity is the first bracket of
  Theorem~\ref{thm:app-factorization-universal-current-interface};
  equivalently, it is the equal-time Poisson bracket smeared against
  \(a(t)b(s)\delta(t-s)\), which produces the pointwise product \(ab\).
\end{proof}

\begin{rmk}[Orientation reversal]
\label{rmk:app-factorization-orientation-reversal}
  The orientation convention is the ordinary convention for oriented
  one-forms and current one-forms.  The labels \(a(t)dt\) and regular
  currents \(B=b(t)dt\) transform by pullback,
  \(a(\psi(s))\psi'(s)\,ds\) and \(b(\psi(s))\psi'(s)\,ds\).  No
  additional sign is inserted from
  \(u_{a(t)dt\otimes f}\), \(c_\lambda\), \(\theta_\lambda\), or the
  principal-part label \(\rho\).  If the written one-form is held fixed
  while only the chosen orientation is reversed, the sign is exactly the
  sign of \(dt\).
\end{rmk}

\begin{rmk}[Boundary of the algebraic current theorem]
\label{rmk:app-factorization-residual}
  Theorem~\ref{thm:app-factorization-universal-current-interface}
  proves the algebraic distributional-current interface.  It does not
  identify \(\Theta_\rho(B)\) with a finite polynomial BV observable,
  construct the primitive cumulant projection before reduction, solve
  the full de Rham-current BV differential, or prove the mixed
  brane-defect QME.  It also does not supply Costello renormalized
  graph weights.  Those statements require the unreduced current
  complex, scalar cancellation data, Tate/weighted coefficient kernel,
  and brane-defect propagator specified in
  Appendix~\ref{sec:app-unreduced-bv-qme} and
  Problem~\ref{prob:analytic-graph-realization}.
\end{rmk}
