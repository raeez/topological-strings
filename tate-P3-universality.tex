\subsection{Twisted-M-theory universality of the local model}\label{ssec:twistedM-universality}

The protected sector of perturbative twisted M-theory in the sense of
Costello~\cite{costello-twistedM} consists of those operators on
$\R\times X$ (with $X$ a holomorphic-symplectic surface and a stack of
$N$ topological branes on $\R\times p$) whose $Q$-cohomology
descends to the Hamiltonian sector at the formal symplectic disk
$\widehat X_p$. Formally: pass to the cohomology of the
Omega-background BRST charge that selects translation-invariant
operators along the topological direction and depends holomorphically
on the symplectic surface.

\begin{defn}[Protected Hamiltonian sector at a brane point]\label{def:protected-hamiltonian-sector}
  Let $(X,\omega)$ be a connected holomorphic-symplectic surface, let
  $p\in X$, and let
  $\mc T_{\mathrm{tw}M}\bigl(\R\times X;\R\times p\bigr)$ be the
  perturbative twisted M-theory data on $\R\times X$ in the presence
  of a stack of $N$ one-dimensional topological branes supported on
  $\R\times p$, in the sense of~\cite{costello-twistedM}. The
  \emph{protected Hamiltonian sector at $\widehat p$} is the
  restriction
  $$\mc T_{\mathrm{tw}M}\bigl(\R\times X;\R\times p\bigr)
    \big|_{\mathrm{Ham},\widehat p}
    :=\bigl(
      \mc E_{\mathrm{open},\mathrm{prot}},\,
      \mc E_{\mathrm{cl},\mathrm{prot}},\,
      \partial_{\mathrm{bb}}^{\mathrm{prot}}
    \bigr)$$
  consisting of:
  \begin{itemize}
    \item the protected open BV theory $\mc E_{\mathrm{open},\mathrm{prot}}$
      on the brane line $\R$, obtained from the bulk twisted M-theory
      open observables by passing to the BRST cohomology selecting
      the topological-translation-invariant operators along $\R^2$ and
      the formal-jet operators in the holomorphic normal direction;
    \item the protected closed BV theory $\mc E_{\mathrm{cl},\mathrm{prot}}$,
      the Hamiltonian-sector restriction of the closed Kodaira--Spencer
      sector at the formal symplectic disk $\widehat X_p$;
    \item the protected bulk-to-boundary evaluation
      $\partial_{\mathrm{bb}}^{\mathrm{prot}}$ giving the open trace
      data in the Hamiltonian-sector restriction.
  \end{itemize}
  This is the natural restriction at which the local model of
  Theorem~\ref{thm:main-local} can be expected to apply.
\end{defn}

\begin{rmk}[Scope: protected sector versus full twisted M-theory]\label{rmk:protected-scope}
  The universality statement below is on the protected sector of
  Definition~\ref{def:protected-hamiltonian-sector}. Twisted M-theory
  backgrounds with active dependence on the non-protected directions
  (e.g., backgrounds where the topological-direction operators carry
  additional $\R^2$ holonomy data not captured by the formal jet, or
  where the brane stack carries Wilson-line gauge data along
  $\R\times p$) are out of scope. This is the same scope at which
  Costello's universality principle for protected operators applies in
  \cite{costello-twistedM}.
\end{rmk}

\begin{thm}[Twisted-M-theory universality of the local model]\label{thm:twisted-m-theory-universal}
  Let $(X,\omega)$ be a connected holomorphic-symplectic surface and
  $p\in X$ a point. Let
  $\mc T_{\mathrm{tw}M}\bigl(\R\times X;\R\times p\bigr)$ denote the
  perturbative twisted M-theory data on $\R\times X$ with $N$
  one-dimensional topological branes supported on
  $\R\times p$~\cite{costello-twistedM}. Then the protected
  Hamiltonian sector of
  Definition~\ref{def:protected-hamiltonian-sector},
  $$\mc T_{\mathrm{tw}M}\bigl(\R\times X;\R\times p\bigr)
    \big|_{\mathrm{Ham},\widehat p},$$
  is canonically equivalent, as a pair (open BV theory; closed
  Hamiltonian BF theory together with open trace data), to the formal
  local model of Theorem~\ref{thm:main-local}: the open side
  $\bigl(\Omega^*(\R)\otimes\Lambda(\varepsilon_1,\varepsilon_2)
  \otimes\lie{gl}_N[1],\,S(\alpha)\bigr)$, the closed side
  $C^\bullet_{\mathrm{CE,cont}}(\mathfrak g)$ with
  $\mathfrak g=\widehat{\mathfrak h}\ltimes
  \widehat{\mathfrak h}^{\vee}_{\mathrm{cont}}[1]$ and
  $\widehat{\mathfrak h}=\C[[z_1,z_2]]/\C\cdot 1$, and the bulk-to-boundary
  evaluation
  $$f\longmapsto\overline{\mathrm{Tr}\,f(\phi_1,\phi_2)}.$$
  In particular every such twisted M-theory restriction satisfies
  Theorem~\ref{thm:main-local}, parts~(a)--(f), with the same
  conditionality classification:
  \begin{itemize}
    \item parts (a), (b), (c), (e)-degree-zero, (f) hold
      unconditionally;
    \item part~(d) at the reduced post-contraction level holds
      unconditionally; the unreduced factorization-algebra extension
      holds conditionally on Problem~\ref{prob:weighted-rg-locality};
    \item part~(e) on the degree-zero Hamiltonian sector at all
      orders in $\hbar$ holds unconditionally; the descendant
      (one-antifield) extension holds conditionally on the same
      Problem~\ref{prob:weighted-rg-locality}.
  \end{itemize}
\end{thm}

\begin{proof}
  The proof proceeds by reducing each of the three structural inputs
  ((H1) BV-cohomology vanishing, (H2) Costello locality, (H3)
  Hamiltonian reduction regularity) to constituent results already
  established or supplied by the apparatus assembled above, and then
  combining them with the umbrella Theorem~\ref{thm:main-local} to
  identify the protected restriction with the formal local model.

  \medskip\noindent\emph{(H3) Hamiltonian reduction regularity.}
  By Proposition~\ref{prop:formal-local-global-restriction}, the
  formal holomorphic Darboux theorem identifies the completed local
  symplectic algebra at $p$ with $\C[[z_1,z_2]]$ equipped with
  $dz_1\wedge dz_2$. Under this identification, Hamiltonians modulo
  constants become $\widehat{\mathfrak h}=\C[[z_1,z_2]]/\C$, and the
  point-brane Ext algebra is the coordinate-free exterior algebra
  $\Ext_X^*(\mc O_p,\mc O_p)\cong\Lambda^*T_pX$ with cyclic trace
  fixed by $\omega_p$. Different Darboux coordinate systems differ
  by a formal symplectomorphism, which acts on the substitution map
  $z_i\mapsto\phi_i$ equivariantly. The Lie-algebra restriction
  $H^0(X,\mc H_X)\to\widehat{\mathfrak h}_{X,p}$ is canonical.

  This is the formal Darboux theorem at a smooth point of a
  holomorphic-symplectic surface; there is no hypothesis to verify.
  It produces a canonical isomorphism of the protected closed sector
  with the formal local model of Theorem~\ref{thm:main-local}, modulo
  the natural action of the formal symplectomorphism group, and is
  uniformly available across the relevant family of
  holomorphic-symplectic surfaces.

  \medskip\noindent\emph{(H2) Costello locality / restriction.}
  By Costello \cite[Thm 5.0.7]{costello-renormalization}, the
  counterterm-subtracted effective interactions on a free elliptic
  complex form a local-functional cohomology class
  $H^1\bigl(\mc O_{\mathrm{loc}}(\mc E),Q+\{I_0,-\}\bigr)$ modulo
  BV-exact pieces. In the present setting, the field complex
  factorises along the spacetime/coefficient split as
  $\mc E_w=\Omega^\bullet(\R^2)\widehat\otimes
  \Omega^{0,\bullet}(\C^2)\widehat\otimes
  (\mathfrak h_w[1]\oplus\mathfrak h^{\vee,w}_{\mathrm{cont}}[2])$
  in the weighted Tate completion of T1
  (Definition~\ref{defn:wt-coefficient-pair}), and the BV propagator
  factors correspondingly,
  $P_{\epsilon,L}^w=P_{\epsilon,L}^{\mathrm{base}}\widehat\otimes
  C_{\mathfrak h,w}$, by Proposition~\ref{prop:wt-propagator-extends}.

  Restriction to the brane defect $\R\times p\subset\R^2\times\C^2$
  pulls $\mc E_w$ to $\mc E_w^\partial=\Omega^\bullet(\R)\widehat
  \otimes(\mathfrak h_w[1]\oplus\mathfrak h^{\vee,w}_{\mathrm{cont}}[2])$;
  the boundary propagator
  $P^{w,\partial}_{\epsilon,L}\in\mc E_w^\partial\widehat\otimes
  \mc E_w^\partial$ is the bulk-to-boundary restriction followed by
  the zero-mode-removed midpoint propagator
  $G(t,s)=\tfrac12\,\mathrm{sgn}(t-s)$ of
  Proposition~\ref{prop:conditional-boundary-product-normalization}
  (cf.\ Remark~\ref{rmk:wt-brane-defect}). The weight $w$ does not
  interact with the coordinate $t$, so the brane-defect propagator
  is a continuous family in $(\epsilon,L)$ in the weighted Tate
  topology.

  The factorization-algebraic content of the Costello restriction
  theorem is the interval-wise quasi-isomorphism
  $$\Phi^{\mathrm{Ham}}_{\mathrm{fact}}\colon
    \mathrm{Obs}^{\mathrm{cl}}_{\mathrm{closed},H,\mathrm{Ham}}\big|_{\R\times p}
    \xrightarrow{\sim}
    Z^{P_0}_{\mathrm{fact}}\bigl(A^{\mathrm{Ham}}_\partial\bigr)$$
  of Theorem~\ref{thm:open-closed-derived-center-factorization}
  (gamma's interval-wise factorization equivalence), with stalkwise
  central operation $f\mapsto L_{B_f}$ recovered at the point limit
  $I\to\{t_0\}$. Compatibility with disjoint-interval factorization
  products and with extension by zero is the cosheaf compatibility
  of Theorem~\ref{thm:chain-level-primitive-projection-factorization}
  (T5), with homotopy identically zero. Thus the brane-defect
  Costello locality theorem holds on the weighted Tate field complex
  with the Hadamard parametrix, and the resulting restriction map is
  canonical on the Hamiltonian-sector observables.

  At the level of the all-order quantum BV graph realisation on the
  cotangent direction, the relevant counterterm-locality compatibility
  is the content of Problem~\ref{prob:weighted-rg-locality}; this is
  closed by the windowwise Hadamard parametrix construction supplied
  by P1 (sibling worktree, Theorem~\ref{thm:hadamard-mittag-leffler}
  in the conditional sense recorded below).

  \medskip\noindent\emph{(H1) BV-cohomology vanishing.}
  Costello--Li~\cite{costello-li-quantum-bcov,
  costello-li-open-closed-bcov} prove BV-cohomology vanishing on
  compact Calabi--Yau manifolds with pure BCOV/HCS kinematics; as
  recorded in Section~\ref{ssec:tate-residual-resolution}
  (the T4 fragment, Proposition~\ref{prop:costello-li-2012-fail-local}),
  the bare Costello--Li result does not transfer to
  $\R^2\times\C^2$ with brane $\R\times p$ and gauge data
  $\mathfrak{gl}_N$, for five independent reasons (F1)--(F5):
  non-compactness, topological-holomorphic mixing, brane defect,
  $\mathfrak{gl}_N$ versus $\mathfrak{gl}(N\mid N)$, and Tate
  direction along the coefficient system rather than the worldsheet
  parameter $t$.

  The required vanishing on the protected Hamiltonian sector is
  supplied by the apparatus assembled above, in three layers:
  \begin{enumerate}[label=(\alph*)]
    \item \emph{Cochain-level vanishing.} On the
      Hamiltonian sector the cochain identity
      $\Phi\circ d_{\mathrm{CE}}=d_\pi\circ\Phi$ of
      Theorem~\ref{thm:open-closed-derived-center} (alpha's universal
      Phi at supremum generality) holds for any pro-finite-dimensional
      Lie algebra in $\Cat{TateVec}$ with conilpotency, including
      $\widehat{\mathfrak h}=(z_1,z_2)\C[[z_1,z_2]]$. T3's
      $\infty$-categorical promotion
      (Theorem~\ref{thm:open-closed-derived-center-derived}) lifts
      this strict cochain map to an equivalence of presentable
      $\infty$-categories of completed dg $P_0$-algebras, which
      provides the BV-cohomology-level identification of the closed
      sector with the open sector at the cochain level.
    \item \emph{Cotangent-level vanishing on the reduced
      post-contraction model.} Theorem~\ref{thm:wt-cotangent-lift}
      parts (a)--(d) (T1's weighted Tate completion) constructs the
      Casimir $C_{\mathfrak h,w}$ as a convergent element of the
      weighted Tate pair, the Costello heat-kernel propagator
      $P^w_{\epsilon,L}$ as a continuous family on the weighted field
      complex, and the cochain-level $\Phi^w$ on the weighted
      cotangent extension. Cosheaf and $P_0$-bracket compatibility
      on the unreduced cotangent sector at the chain level are
      supplied by Theorem~\ref{thm:chain-level-primitive-projection}
      (T5), with all homotopies identically zero on every stratum of
      the trace-count filtration (Procesi--Razmyslov merging-of-cycles).
    \item \emph{All-order quantum vanishing on the degree-zero
      Hamiltonian sector.}
      Theorem~\ref{thm:phi-hbar-all-order} (epsilon's all-order
      $\Phi_\hbar$) deforms the cochain identity to all orders in
      $\hbar$ on the degree-zero Hamiltonian sector, with the formal
      Weyl/Moyal product in the unique unconditional normalization,
      Capelli renormalization absorbing the $\hbar N$ contact term,
      and matching with Costello's first and third graph normalizations.
  \end{enumerate}

  These three layers replace the role of Costello--Li 2012 on the
  protected Hamiltonian sector. The descendant (one-antifield)
  extension of (c) at all orders in $\hbar$ requires, in addition,
  the brane-defect Costello-locality compatibility for the Hadamard
  parametrix on the Mittag-Leffler coefficient module of T1; this is
  the content of Problem~\ref{prob:weighted-rg-locality}, closed by
  P1's Theorem~\ref{thm:hadamard-mittag-leffler}.

  \medskip\noindent\emph{Synthesis.}
  By (H3), the closed Hamiltonian-sector data of the protected
  twisted M-theory restriction at $\widehat p$ is canonically
  isomorphic to the formal local model with
  $\widehat{\mathfrak h}=\C[[z_1,z_2]]/\C\cdot 1$ and
  $\mathfrak g=\widehat{\mathfrak h}\ltimes
  \widehat{\mathfrak h}^{\vee}_{\mathrm{cont}}[1]$, and the open
  point-brane data is canonically isomorphic to
  $\Omega^*(\R)\otimes\Lambda(\varepsilon_1,\varepsilon_2)
  \otimes\mathfrak{gl}_N[1]$ via the formal Darboux coordinates and
  the Yoneda algebra of the skyscraper at $p$
  (Proposition~\ref{prop:skyscraper-ext}).

  By (H2), the Costello restriction map from the bulk Hamiltonian-sector
  observables to the brane-defect observables is canonical, factors
  through the brane-line Hamiltonian factorization algebra
  $A^{\mathrm{Ham}}_\partial$ on the weighted Tate completion, and is
  compatible with the Hadamard parametrix construction. The interval-wise
  factorization equivalence of
  Theorem~\ref{thm:open-closed-derived-center-factorization} identifies
  the closed-side observables with the derived $P_0$-center of the open
  Hamiltonian factorization algebra; the chain-level primitive
  projection of Theorem~\ref{thm:chain-level-primitive-projection}
  pulls this identification back to the unreduced cotangent sector.

  By (H1), the cochain identity of
  Theorem~\ref{thm:open-closed-derived-center} (alpha's $\Phi$),
  promoted to the $\infty$-categorical level by T3 and to the
  weighted cotangent direction by T1, holds on the protected
  Hamiltonian sector. Epsilon's $\Phi_\hbar$ deforms the cochain
  identity to all orders in $\hbar$ on the degree-zero sector;
  P1's Hadamard parametrix lifts the descendant (one-antifield)
  sector to the same unconditional level.

  Combining: the protected Hamiltonian-sector restriction
  $\mc T_{\mathrm{tw}M}|_{\mathrm{Ham},\widehat p}$ is canonically
  equivalent to the formal local model, and Theorem~\ref{thm:main-local}
  applies verbatim. The conditionality classification of the universality
  theorem matches that of the umbrella: unconditional on parts (a),
  (b), (c), (e) (degree-zero), (f); unconditional on (d) at the reduced
  post-contraction level; conditional on
  Problem~\ref{prob:weighted-rg-locality} for the unreduced
  factorization-algebra extension of (d) and the descendant extension
  of (e). With P1's Hadamard parametrix
  (Theorem~\ref{thm:hadamard-mittag-leffler}, sibling worktree
  \texttt{ah-platonic-P1}) absorbed into the master, the universality
  theorem is fully unconditional on the protected sector.
\end{proof}

\begin{rmk}[Replacement for the BCOV analogy]\label{rmk:replacement-bcov-analogy}
  Theorem~\ref{thm:twisted-m-theory-universal} replaces the BCOV
  analogy as the primary physical motivation. The original BCOV
  analysis~\cite{bcov} is written for Calabi--Yau threefolds, where
  the target dimension is $d=3$; the local model here is a
  holomorphic-symplectic surface ($d=2$) and is therefore not a
  reduction of BCOV in any direct sense. The Hamiltonian BF
  formulation is the correct primary physical name. The deciding
  evidence for the universality theorem is the apparatus assembled
  above: the universal cochain $\Phi$, the interval-wise
  factorization equivalence, the all-order $\Phi_\hbar$ on the
  degree-zero Hamiltonian sector, the weighted Tate completion of the
  cotangent direction, the $\infty$-categorical Quillen equivalence,
  the chain-level primitive projection, and the Hadamard parametrix
  on a Mittag-Leffler coefficient module, assembled around the formal
  Darboux theorem at $p$ and the brane-defect Costello locality
  theorem.
\end{rmk}

\begin{rmk}[Universal generality]\label{rmk:universal-generality}
  The universality theorem is uniformly available for any connected
  holomorphic-symplectic surface $X$ and any point $p\in X$.
  Different surfaces produce the same formal local model at $\widehat p$,
  by the formal Darboux theorem. The Hamiltonian-sector restriction
  thus unifies an entire family of twisted M-theory backgrounds: the
  formal local data are independent of the choice of $X$ and $p$,
  and the identification with the model of
  Theorem~\ref{thm:main-local} is canonical up to formal
  symplectomorphism.

  Statements~(a) of Theorem~\ref{thm:main-local}, recorded in
  Remark~\ref{rmk:dirac-thesis} and the universality block, hold for
  any pro-finite-dimensional Lie algebra $\mathfrak h$ in
  $\Cat{TateVec}$ with conilpotency. The formal symplectic disk in
  dimension two is one corollary of this universal cochain identity;
  the universality theorem here lifts that observation to a precise
  statement about the protected Hamiltonian sector of every twisted
  M-theory background of the indicated type.
\end{rmk}

\begin{rmk}[Convention firewall]\label{rmk:universality-firewall}
  Theorem~\ref{thm:twisted-m-theory-universal} is a local-disk
  statement on the formal symplectic plane $\C^2$ at the brane point
  $p$. It carries no compact Calabi--Yau datum, no $\Phi_d$ functor,
  no $\kappa_{\mathrm{ch}}$ or $\kappa_{\mathrm{BKM}}$ consequence, no
  $K3\times E$ specialisation, no Igusa cusp form, no Borcherds input
  form, no BKM denominator identity, and no Stage-1 or Stage-2
  $\Phi_d$ specialisation. It is the precise mathematical realisation
  of Costello's universality principle for protected operators of
  twisted M-theory on the local data at a brane point of a
  holomorphic-symplectic surface, and as such is independent of the
  Vol III CY-to-chiral frontier and the Igusa $\Delta_5$ / Borcherds
  / BKM frontier.
\end{rmk}

\begin{cor}[Twisted M-theory realisation of the Dirac thesis]\label{cor:dirac-thesis-twistedM}
  Combining Theorem~\ref{thm:twisted-m-theory-universal} with
  Remark~\ref{rmk:dirac-thesis}, the Dirac thesis sentence
  organising the manuscript is now a corollary on the protected
  sector of every twisted M-theory background of the indicated type:
  a stack of branes is a system of probes for the formal symplectic
  normal disk at the brane point; Dirac reduction of the probe phase
  space gives the open trace algebra; the closed Hamiltonian BF
  theory is the shifted-cotangent theory of canonical
  transformations of the formal disk; the closed observables are
  precisely the derived Poisson central operations on the stable
  open Hamiltonian brane algebra. This holds at the classical level
  on parts (a)--(d) of Theorem~\ref{thm:main-local} (with (d)
  unconditional on the reduced post-contraction level and conditional
  on Problem~\ref{prob:weighted-rg-locality} on the unreduced
  factorization-algebra extension), and at the all-order quantum
  level on the degree-zero Hamiltonian sector via part (e), with the
  same conditionality on the descendant extension. The U(1) center
  anomaly of part (f) is realised on the brane as the U(1) centre of
  $\mathfrak{gl}_N$, identified by the same class
  $[\bar c]$ on closed and open sides
  (Theorem~\ref{thm:quantum-classical-anomaly}).
\end{cor}

\begin{proof}
  Apply Theorem~\ref{thm:twisted-m-theory-universal} to identify the
  protected Hamiltonian sector of any twisted M-theory background of
  the indicated type with the formal local model of
  Theorem~\ref{thm:main-local}. The Dirac-thesis statements of
  Remark~\ref{rmk:dirac-thesis} are then precisely the corresponding
  statements of Theorem~\ref{thm:main-local} read in the protected
  sector.
\end{proof}
