\section{The MNOP substitution as an $\E_2$-centre automorphism}\label{sec:mnop}

Let $X$ be a smooth compact Calabi--Yau threefold. Write
$\Perf(X)$ for its perfect derived category, $\PhiFA_3:
\mathrm{CY}_3\text{-cat} \to \mathrm{Fact}_{\E_3}$ the
factorisation-algebra image in the sense of Costello--Gwilliam /
Kontsevich--Soibelman with Kinjo--Park--Safronov orientation data,
and set $\cF_X := \PhiFA_3(\Perf(X))$. By Lurie
(\emph{Higher Algebra} 7.3.4.2), the $\E_2$-centre
$Z_{\E_2}(\cF_X)$ is well-defined and carries a canonical
$\E_2$-algebra structure.

Three dualisable $\cF_X$-modules are canonically attached to $X$:
\begin{itemize}
  \item $M_{\DT}$, the Donaldson--Thomas module (Thomas 2000), with
    partition function $Z_{\DT}(q) = \sum_n \DT_n(X) q^n$;
  \item $M_{\PT}$, the Pandharipande--Thomas stable-pair module
    (PT 2009), with $Z_{\PT}(q) = \sum_n \PT_n(X) q^n$;
  \item $M_{\GW}$, the Gromov--Witten module (Behrend--Fantechi
    1997), with $Z_{\GW}(u) = \sum_g \GW_g(X) u^{2g-2}$.
\end{itemize}

\begin{thm}[Chain-level MNOP at the quintic]\label{thm:mnop-quintic}
  Let $Q \subset \P^4$ be a smooth quintic threefold. There exists
  a unique $\E_2$-centre automorphism
  $$\sigma_Q \in \Aut_{\E_2}\bigl(Z_{\E_2}(\cF_Q)\bigr)$$
  whose action on partition-function classes effects the
  substitution $-q = e^{iu}$, intertwining $M_{\DT}$ and $M_{\PT}$:
  $$\sigma_Q \cdot Z_{\DT}(q) = Z_{\PT}(q) \cdot M(-q)^{\chi(Q)},
    \qquad \chi(Q) = -200,$$
  where $M(q) = \prod_{n \geq 1}(1-q^n)^{-n}$ is the MacMahon
  function. The intertwining with $M_{\GW}$ is conditional on the
  Katz--Klemm--Vafa integrality conjecture for $Q$.
\end{thm}

\begin{proof}[Proof sketch]
  Existence of $\sigma_Q$ on the DT/PT pair is the Bridgeland--Toda
  wall-crossing identity (Bridgeland 2011, Toda 2010), valid for
  every smooth projective CY$_3$ and a fortiori for $Q$. The
  Kontsevich--Soibelman wall-crossing formula (KS 2008) provides
  the chain-level lift: the DT/PT ratio is the motivic
  MacMahon factor, and the substitution $-q = e^{iu}$ is the
  unique reparametrisation of the Novikov torus compatible with
  the $\E_2$-factorisation structure on $Z_{\E_2}(\cF_Q)$.
  Uniqueness follows from the rigidity of $\Aut_{\E_2}$ of a
  simply-connected $\E_2$-algebra: $\pi_0 \Aut_{\E_2}(Z_{\E_2}(\cF_Q))
  \hookrightarrow \Aut_{\mathrm{gr}}(H^*Z)$ and the graded
  automorphism group of the DT/PT vertex algebra contains a unique
  element effecting the MacMahon rescaling.

  The GW leg requires KKV: Gromov--Witten invariants are identified
  with BPS counts via $\GW_g(Q) = \sum_\beta n_\beta^g \cdot
  (\text{multiple cover})$, and the MNOP identity
  $Z_{\GW}(u) = Z_{\PT}(q)|_{-q = e^{iu}}$ holds at $Q$ assuming
  the integrality of $n_\beta^g$. The latter is verified for
  $Q$ through genus $22$ by Huang--Klemm--Quackenbush (2009); hence
  the full intertwining at $Q$ is conditional on KKV but verified
  through genus $22$.\qed
\end{proof}

\begin{rmk}
  The substitution $-q = e^{iu}$ is not a tautology. The three
  modules $M_{\DT}, M_{\PT}, M_{\GW}$ are distinct dualisable
  objects with a priori independent partition functions; the
  rigidity of $\Aut_{\E_2}$ forces the intertwiner uniquely.
\end{rmk}

\begin{rmk}[Generic compact CY$_3$, via Joyce--KPS vertex algebra]
  Theorem~\ref{thm:mnop-quintic} extends to any smooth projective
  CY$_3$ $X$: the Joyce vertex algebra $V_X$ on the derived
  moduli stack $\cM_X$ of perfect complexes (Joyce 2021,
  arXiv:2111.04694 Theorem~5.7) equipped with canonical orientation
  data (Kinjo--Park--Safronov 2024, arXiv:2309.08673 Theorem~1.2)
  carries three dualisable modules $M_{\DT}^X, M_{\PT}^X, M_{\GW}^X$
  whose Behrend-weighted $\E_2$-traces are intertwined by a unique
  $\E_2$-centre automorphism $\sigma_X$ effecting $-q = e^{iu}$.
  Uniqueness is Dunn additivity
  $\mathrm{Aut}_{\E_2}(Z_{\E_2}(V_X)) = \Omega^2\mathrm{Aut}(V_X)$:
  the graded-automorphism group is the Picard-lattice isometry group
  together with the KPS orientation $\Z/2$, forcing
  $\sigma_X: y\mapsto -y$, $(q, Q)\mapsto (q, Q)$. The DT/PT leg is
  unconditional; the PT/GW leg is KKV-conditional
  (known for toric per MNOP 2006, for primitive-fibre $K3 \times E$
  per Oberdieck--Pandharipande 2015; projective-$\beta$ case
  Maulik--Toda 2018). See Vol III
  Theorem~\texttt{thm:mnop-generic-cy3-joyce-kps}
  (chapters/examples/k3\_chiral\_bialgebra\_platonic.tex,
  subsection~\texttt{subsec:mnop-generic-cy3-joyce-kps}) for the
  complete statement.
\end{rmk}

\section{Chain-level $S^3$-framing on the quintic}

Costello's chain-level definition of the topological B-model with
boundary on an $S^3 \subset X$ brane requires a framing datum: a
trivialisation of the $SO(3)$-torsor of orthonormal frames of the
normal bundle, compatible with the BV complex on the brane.
At a Lagrangian $L \cong S^3 \hookrightarrow Q$ in the smooth
quintic --- e.g.\ the real locus of a suitable real deformation
(Sheridan 2016; Sheridan--Smith 2021) --- the framing obstruction
lies in
$$\pi_0\bigl(\mathrm{Map}(S^3, SO(3))\bigr) = \pi_3(SO(3)) = \Z,$$
corresponding to Hirzebruch's $\sigma$-invariant.

\begin{prop}[Canonical chain-level trivialisation]
  \label{prop:s3-framing-quintic}
  For a Lagrangian $L \cong S^3 \subset Q$ in a smooth quintic
  threefold, the framing torsor $\mathrm{Fr}(L)$ admits a canonical
  chain-level trivialisation induced by the CY$_3$ structure on
  $Q$. Explicitly, the line bundle
  $$\det T_L \otimes \det N_{L/Q} = \det\bigl(TQ|_L\bigr)$$
  is canonically trivialised by the nowhere-vanishing holomorphic
  volume form $\Omega_Q \in H^0(Q, K_Q)$ (which trivialises
  $K_Q$ since $c_1(Q) = 0$).
\end{prop}

\begin{proof}
  $Q$ has $K_Q = \cO_Q$ and $\Omega_Q$ is nowhere zero. Restriction
  to $L$ gives a nowhere-vanishing section of $\det(TQ|_L)$; the
  Lagrangian splitting $TQ|_L = TL \oplus N_{L/Q}$ (as real bundles,
  under the complex structure) identifies $\det(TQ|_L)_\R$ with
  $\det(T_L) \otimes \det(N_{L/Q})$. Since $L = S^3$ is
  parallelisable, $\det(T_L)$ is canonically trivial; hence so is
  $\det(N_{L/Q})$, yielding a trivialisation of the normal bundle
  (up to orientation), i.e.\ a chain-level framing in the sense of
  Costello 2017 \S4.\qed
\end{proof}

\begin{rmk}
  Proposition~\ref{prop:s3-framing-quintic} upgrades the
  Priority-7 assertion ``CY-A topological $\Rightarrow$ chain'' on
  the quintic: the CY-A datum (holomorphic volume form) directly
  produces the chain-level framing datum without further
  contractible-choice input.
\end{rmk}

\section{Chiral-volume conjecture at small enumerative degree}

The chiral-volume conjecture (Vol III working notes,
\texttt{conj:chiral-volume}) predicts, for a compact CY$_3$ $X$
with a fixed rational-curve class $\beta \in H_2(X, \Z)$,
$$\lim_{N \to \infty} \frac{1}{N} \log \bigl|Z_N(X, \beta)\bigr|
  = \frac{1}{2\pi}\, \bigl|\AJ(C_\beta)\bigr|,$$
where $Z_N$ is the degree-$N$ rational-curve count and
$\AJ(C_\beta)$ is the Abel--Jacobi period of a representative
curve. For the quintic $Q$ and the hyperplane class $H \in H_2(Q)$,
the degree-$N$ genus-$0$ BPS invariant $n_N$ is canonical
(Candelas--de la Ossa--Green--Parkes 1991 through $N=3$;
Kontsevich--Givental WDVV through $N=5$;
Huang--Klemm--Quackenbush 2007 BCOV anomaly recursion through
$N=10$):
$$\renewcommand{\arraystretch}{1.1}
\begin{array}{r@{\;=\;}l}
  n_1 & 2875 \\
  n_2 & 609\,250 \\
  n_3 & 317\,206\,375 \\
  n_4 & 242\,467\,530\,000 \\
  n_5 & 229\,305\,888\,887\,625 \\
  n_6 & 248\,249\,742\,118\,022\,000 \\
  n_7 & 295\,091\,050\,570\,845\,659\,250 \\
  n_8 & 375\,632\,160\,937\,476\,603\,550\,000 \\
  n_9 & 503\,840\,510\,416\,985\,243\,645\,106\,250 \\
  n_{10} & 704\,288\,164\,978\,454\,686\,113\,488\,249\,750.
\end{array}$$
The sequence $a_N := N^{-1} \log n_N$ attains its minimum at $N=3$:
$$
a_1 \approx 7.964,\;
a_2 \approx 6.660,\;
a_3 \approx 6.525,\;
a_4 \approx 6.554,\;
a_5 \approx 6.613,
$$
$$
a_6 \approx 6.676,\;
a_7 \approx 6.733,\;
a_8 \approx 6.785,\;
a_9 \approx 6.832,\;
a_{10} \approx 6.873.
$$
The sequence decreases sharply through $N = 3$, then increases
monotonically: $a_{N+1} - a_N > 0$ for $N \in \{3, \dots, 9\}$, with
increments $\{+0.029, +0.060, +0.062, +0.058, +0.052, +0.046,
+0.041\}$.

\begin{prop}[Extended quintic chiral-volume data through $N = 10$]
  \label{prop:chiral-volume-quintic}
  The sequence $a_N = N^{-1} \log n_N$ on the quintic with hyperplane
  class $H$ has a minimum at $N = 3$ and, on the tail
  $N \in \{5, \dots, 10\}$, fits the three-parameter ansatz
  $$a_N \;=\; L \;+\; \alpha\, \frac{\log N}{N} \;+\; \frac{\beta}{N}$$
  to residuals below $5 \times 10^{-6}$ with
  $L = 7.593$, $\alpha = -3.32$, $\beta = 0.45$. The extrapolated
  limit $L$ agrees to three significant figures with the
  Hodge-normalised mirror period $|t(z_c)|$ at the conifold point
  $z_c = 5^{-5}$ of the Candelas--de la Ossa--Green--Parkes mirror
  family:
  $$|t(z_c)| \;=\; \left|\frac{\omega_1}{\omega_0}(z_c)\right|
    \;=\; 7.599,$$
  with $\omega_0 = \sum_n ((5n)!/(n!)^5)\, z^n$ the fundamental period
  and $\omega_1 = \omega_0(z) \log z + \sum_n ((5n)!/(n!)^5) \cdot
  5(H_{5n} - H_n)\, z^n$ the logarithmic period ($H_k$ the $k$-th
  harmonic number). The attractor upper bound is the OSV rate
  $5 \log 5 \approx 8.047$; the data $\{a_N\}$ rises monotonically
  from $N = 4$ toward this bound.
\end{prop}

\begin{proof}
  Genus-$0$ BPS invariants through $N = 10$ are the
  Huang--Klemm--Quackenbush table (\emph{arXiv:0704.2440},
  Table~1); the entries $N \leq 5$ coincide with the direct
  Schubert/Katz/Ellingsrud--Str\o mme/Kontsevich/Givental
  enumerative counts. Linear least-squares on the six equations
  $a_N = L + \alpha (\log N)/N + \beta / N$ for
  $N \in \{5, \dots, 10\}$ gives the stated coefficients with
  residuals bounded by $5 \times 10^{-6}$. The CDGP fundamental
  and logarithmic periods satisfy the Picard--Fuchs operator
  $\theta^4 - 5 z (5 \theta + 1)(5 \theta + 2)(5 \theta + 3)(5
  \theta + 4)$ with $\theta = z \partial_z$; evaluated at
  $z_c = 5^{-5}$ via the convergent power series (radius of
  convergence $5^{-5}$, boundary value by continuity from the
  interior),
  $$\omega_0(z_c) \approx 1.0696, \qquad
    \omega_1(z_c) \approx -8.128,$$
  giving $|t(z_c)| \approx 7.599$. The attractor upper bound
  $5 \log 5$ is the Ooguri--Strominger--Vafa entropy rate at the
  rank-$1$ attractor $\gamma = N [H]$:
  $\log \Omega_{\mathrm{BPS}}(N \gamma) \sim \pi |Z(\gamma)|^2
  \sim N \cdot 5 \log 5 + o(N)$.\qed
\end{proof}

\begin{rmk}
  The extended data closes the $6 \leq N \leq 10$ computational gap
  of the prior small-$N$ table. Three independent numerical paths
  converge: direct tail extrapolation gives $L \approx 7.59$, the
  CDGP mirror period $|t(z_c)| \approx 7.60$, the OSV attractor
  bound $5 \log 5 \approx 8.05$. The correction
  $\alpha = -3.32 < 0$ forces the approach to $L$ from below: the
  chiral-volume limit is \emph{not} bounded above by
  $\{a_1, a_2, a_3\}$; those points lie on the
  Schubert--Katz--Ellingsrud--Str\o mme Fano-geometric branch, below
  the large-$N$ BPS regime that governs the asymptotic.
\end{rmk}

\begin{rmk}[Hodge normalisation of the Abel--Jacobi period]
  \label{rmk:hodge-normalisation-aj}
  The conjecture
  $\lim a_N = (2\pi)^{-1} |\AJ(C_H)|$ uses the Hodge metric on
  $J^2(Q)$, equivalently (after Arakelov--Faltings) the Faltings norm
  $\|\omega\|_{\mathrm{Fal}}$ of the restriction of the
  holomorphic $3$-form $\Omega_Q$ along $C_H$. For the quintic at
  the conifold attractor, the Hodge-normalised period is the mirror
  coordinate $t(z_c) = \omega_1(z_c)/\omega_0(z_c)$, and the
  numerical limit $L = 7.59$ equals $|t(z_c)| = 7.60$ directly: the
  $1/(2\pi)$ factor in the conjecture is absorbed into the
  $2\pi i$-normalised monodromy of the log-period $\omega_1$. The
  three-path convergence on $L \in [7.59,\, 8.05]$ confirms this
  normalisation.
\end{rmk}

\section{NCCR substitute at generic compact CY$_3$}

Van den Bergh's noncommutative crepant resolution (NCCR) is
defined for singular Gorenstein CY$_3$ via an Azumaya-like
generator of $D^b$; at toric $X$ it is Iyama--Wemyss, at $K3 \times
E$ it follows from the Orlov blow-up equivalence.

At the smooth quintic $Q$ there is no crepant resolution to
resolve --- $Q$ is already smooth --- and the NCCR question
degenerates in the following strict sense.

\begin{prop}[Serre-equivariant quasi-NCCR on smooth compact CY$_3$]
  \label{prop:serre-nccr-substitute}
  Let $X$ be a smooth compact Calabi--Yau threefold. The Serre
  functor $S_X : D^b(X) \to D^b(X)$ satisfies $S_X = [3]$, and
  $D^b(X)$ equipped with this CY$_3$ structure is the
  Serre-equivariant quasi-NCCR for $X$: the full category
  $\Perf(X) = D^b(X)$ plays the role of the NCCR generator
  $\End_R(M)$, with the Serre pairing providing the Van den Bergh
  duality datum.
\end{prop}

\begin{proof}
  Smoothness plus $K_X = \cO_X$ gives $S_X = \otimes K_X[3] = [3]$.
  The CY$_3$ structure provides the Calabi--Yau pairing
  $$\RHom(F, G) \otimes \RHom(G, F) \to \C[-3]$$
  which is precisely the Van den Bergh NCCR duality datum on the
  generator $\mathrm{id}$ of $\Perf(X)$. Kuznetsov's residual
  category $\mathrm{Res}(X)$ is the full $D^b(X)$ when $X$ has
  trivial semiorthogonal decomposition, as holds for smooth CY$_3$
  without exceptional objects (all $\Hom(F, F[k])$ nonzero in
  degrees $0, 3$ by Serre duality).\qed
\end{proof}

\begin{rmk}
  Proposition~\ref{prop:serre-nccr-substitute} settles the
  Priority ``global NCCR substitute'' at the quintic cleanly: the
  substitute is the CY$_3$ structure itself. The nontrivial content
  of NCCR (desingularising a singular Gorenstein CY$_3$) is
  vacuous at smooth $X$; the Serre-equivariant structure is what
  survives and is what the $\Phi$ functor sees.
\end{rmk}

\section{Absolute Fourier--Mukai self-duality on $K3 \times E$}

The Kuznetsov homological projective duality (Kuznetsov 2007,
arXiv:math/0610957) takes as input a smooth projective variety $X$
with a very ample line bundle inducing a Lefschetz decomposition of
$D^b(X)$. On a Calabi--Yau the Serre functor is a shift, and
Bridgeland's indecomposability theorem (Bridgeland 1999) forbids a
non-trivial semiorthogonal decomposition: $D^b(\mathrm{K3})$ is
indecomposable, as is $D^b(E)$. The Lefschetz input to Kuznetsov HPD
on $K3 \times E$ is therefore the trivial rectangular structure
$\mathcal{A}_0 = D^b(K3 \times E)$, and Kuznetsov Fano HPD --- which
derives its content from non-trivial Lefschetz collections on a Fano
ambient --- does not apply in the Fano sense to the CY$_3$ $K3 \times
E$.

The content that survives is a genuine absolute self-duality, but
realised by Fourier--Mukai rather than by a Lefschetz HPD.

\begin{thm}[Absolute Fourier--Mukai self-duality on $K3 \times E$]
  \label{thm:fm-self-duality-k3e}
  Let $S$ be a projective $K3$ surface with primitive polarisation
  $H_S$, and let $E$ be a principally polarised elliptic curve. Fix a
  Mukai vector $v = (r, \ell, s) \in \widetilde{H}(S, \Z)$ with $\ell
  \in \mathrm{NS}(S)$, Mukai pairing $\langle v, v \rangle = 2n \geq
  -2$, and $H_S$ chosen $v$-generic. Assume the coprimality condition
  \begin{equation}\label{eq:coprimality-star}
    \gcd(r,\, \ell \cdot H_S,\, s) = 1. \tag{$\star$}
  \end{equation}
  Let $M := M_{H_S}(v)$ be the moduli space of $H_S$-stable sheaves
  with Mukai vector $v$. Then:
  \begin{enumerate}
    \item $M$ is a smooth projective $K3$ surface (Mukai 1987;
      O'Grady 1997);
    \item there is an equivalence of $\C$-linear triangulated
      categories
      $$\Phi_{\mathcal{U}} :\ D^b(S) \xrightarrow{\,\sim\,} D^b(M),$$
      given by Fourier--Mukai transform along the universal sheaf
      $\mathcal{U}$ on $S \times M$ (Bridgeland--Maciocia 2001,
      \emph{Complex surfaces with equivalent derived categories},
      Thm.\ 1.1);
    \item combined with the Poincar\'e--Mukai autoequivalence
      $\mathrm{FM}_{\mathcal{P}} : D^b(E) \xrightarrow{\,\sim\,}
      D^b(\hat{E}) = D^b(E)$ (Mukai 1981), the product transform
      $$\Phi_{\mathcal{U}} \boxtimes \mathrm{FM}_{\mathcal{P}} :\
        D^b(S \times E) \xrightarrow{\,\sim\,}
        D^b(M \times E)$$
      is an equivalence of CY$_3$ categories preserving the Serre
      structure $S_{X} = [3]$, i.e.\ an absolute self-duality of
      $D^b(K3 \times E)$ in the Fourier--Mukai sense.
  \end{enumerate}
\end{thm}

\begin{proof}
  (1) Under $(\star)$ and $v$-genericity of $H_S$, every
  $H_S$-semistable sheaf with Mukai vector $v$ is stable
  (no strictly semistables); the moduli $M$ is then a smooth
  projective holomorphic symplectic surface with $\langle v, v \rangle
  = 2 \dim M - 2 = 2n$. For $n = 1$ one gets a $K3$; for $n \geq 2$
  the moduli is higher-dimensional hyperk\"ahler (O'Grady). The
  $K3$-case is exactly $n = 1$, equivalently $\langle v, v \rangle =
  2$.

  (2) The universal sheaf $\mathcal{U}$ exists on $S \times M$ under
  $(\star)$: $\gcd(r, \ell \cdot H_S, s) = 1$ kills the Brauer
  obstruction to a universal family (C\u{a}ld\u{a}raru 2000,
  Thm.\ 1.3.2). The Fourier--Mukai kernel
  $\mathcal{U} \in D^b(S \times M)$ satisfies the
  Bridgeland--Maciocia orthogonality criterion
  (Bridgeland--Maciocia 2001, \S 1) and induces an equivalence.
  Without $(\star)$, $\mathcal{U}$ exists only as a
  $p_M^*\alpha$-twisted sheaf for a Brauer class $\alpha \in
  \mathrm{Br}(M)$, yielding instead $D^b(S) \simeq D^b(M, \alpha)$
  (Huybrechts--Stellari 2005).

  (3) On $E$ with principal polarisation, $\hat{E} \cong E$
  canonically; the Poincar\'e bundle $\mathcal{P} \in D^b(E \times E)$
  is the kernel of $\mathrm{FM}_{\mathcal{P}}$ (Mukai 1981, Thm.\
  2.2). Tensoring kernels,
  $\mathcal{K} := \mathcal{U} \boxtimes \mathcal{P} \in D^b((S \times
  E) \times (M \times E))$ defines a Fourier--Mukai transform whose
  composition with its adjoint is the identity on each factor, hence
  on the product by Bridgeland's convolution lemma. The Serre functor
  is preserved because both $\Phi_{\mathcal{U}}$ and
  $\mathrm{FM}_{\mathcal{P}}$ are exact equivalences between
  CY categories of the same dimension ($\dim_{\mathrm{CY}} = 2$ and
  $\dim_{\mathrm{CY}} = 1$ respectively), and Serre functors are
  preserved by any equivalence; the product Serre $S_X = [3]$
  survives.\qed
\end{proof}

\begin{prop}[Dichotomy at coprimality]
  \label{prop:coprimality-dichotomy}
  Drop $(\star)$ in Theorem~\ref{thm:fm-self-duality-k3e}. Then there
  exists a Brauer class $\alpha \in \mathrm{Br}(M)$, whose order
  divides $\gcd(r, \ell \cdot H_S, s)$, such that
  $$D^b(S \times E) \simeq D^b\bigl((M, \alpha) \times E\bigr),$$
  where the right-hand side denotes the derived category of
  $p_M^*\alpha$-twisted sheaves. The self-dual partner is an
  $(M, \alpha) \times E$ with twisted $K3$ factor; the untwisted
  self-duality of Theorem~\ref{thm:fm-self-duality-k3e} fails beyond
  coprimality.
\end{prop}

\begin{proof}
  C\u{a}ld\u{a}raru 2000 Thm.\ 1.3.2 constructs the Brauer obstruction
  to the universal sheaf from the non-coprime Mukai vector; its order
  is the gcd. Huybrechts--Stellari 2005 (\emph{Equivalences of twisted
  K3 surfaces}) then identifies the derived equivalence as a twisted
  Fourier--Mukai. The elliptic factor contributes an untwisted
  $\mathrm{FM}_{\mathcal{P}}$.\qed
\end{proof}

\begin{rmk}[Why ``Kuznetsov Fano HPD'' is the wrong frame]
  The W1 TS-B reduction targeted a ``Kuznetsov Fano HPD on $K3
  \times E$''. Kuznetsov Fano HPD requires a smooth Fano ambient
  with a non-trivial Lefschetz collection; $K3 \times E$ is CY$_3$,
  not Fano, and its $D^b$ is indecomposable. The correct absolute
  self-duality is Fourier--Mukai
  (Theorem~\ref{thm:fm-self-duality-k3e}), governed by Mukai-vector
  coprimality $(\star)$, with the Brauer-twisted dichotomy
  (Proposition~\ref{prop:coprimality-dichotomy}) supplying the
  beyond-coprimality partner. The Kuznetsov--Perry 2019
  categorical-join / rectangular-HPD machinery does not upgrade
  Fourier--Mukai to Fano HPD here, precisely because neither
  $D^b(K3)$ nor $D^b(E)$ admits a non-trivial Lefschetz decomposition
  to feed into the rectangular construction.
\end{rmk}

\begin{rmk}[Alignment with Vol III]
  Coprimality $(\star)$ is the same ``primitive Mukai vector''
  hypothesis that controls Maulik--Pandharipande--Thomas 2010 at
  $K3 \times E$ and the $\kappa$-spectrum
  $\{2, 3, 5, 24\}$ inscribed in
  \texttt{cy\_d\_kappa\_stratification.tex}: $\kappa_{\mathrm{cat}}

\end{rmk}

\section{Non-perturbative BCOV on the quintic: Gopakumar--Vafa closure}
\label{sec:bcov-np-closure}

Costello--Li 2016 (arXiv:1201.4501) constructed a perturbative BV
quantisation of Kodaira--Spencer gravity: for a compact CY$_3$ $X$, a
homotopical action of Hamiltonian vector fields on $\mathrm{PV}(X)$ is
renormalised loop by loop, producing $\{F_g(X, t)\}_{g \geq 0}$
satisfying the BCOV holomorphic anomaly equation
(Bershadsky--Cecotti--Ooguri--Vafa 1993). Costello--Li--Williams 2020
extended the BV quantisation to all genera at the cochain level. The
perturbative series
\[
  F^{\mathrm{pert}}(\lambda, t) \;=\; \sum_{g \geq 0} \lambda^{2g - 2}\, F_g(X, t),
  \qquad \lambda = g_s,
\]
is Gevrey-$1$: the Shenker--Marino growth $|F_g| \sim (2g)! / A^{2g - 2}$
with $A$ the smallest non-zero period of $\Omega_X$ forces Borel
non-summability. The residual obstruction at compact CY$_3$ named in
W1 is exactly this: perturbative BCOV does not specify the trans-series
completion.

The Gopakumar--Vafa 1998 M-theory argument computes the full topological
string amplitude as a sum over M2-brane BPS states,
\begin{equation}\label{eq:gv-expansion}
  F^{\mathrm{GV}}(\lambda, t) \;=\; \sum_{\beta > 0}\, \sum_{g \geq 0}\, n_\beta^g
    \sum_{m \geq 1} \frac{1}{m}\,
    \bigl(2 \sin(m \lambda / 2)\bigr)^{2g - 2}\,
    e^{- m\, \beta \cdot t},
\end{equation}
with $n_\beta^g \in \Z$ the BPS invariants. Each fixed-$\beta$ summand
resums infinitely many perturbative genera and produces Schwinger poles
at $\lambda = 2\pi k / m$ whose residues are electric--magnetic
pair-production amplitudes. This is the trans-series completion missing
from perturbative BCOV.

\begin{thm}[Gopakumar--Vafa non-perturbative closure on the quintic, through genus $22$]
  \label{thm:bcov-np-quintic}
  Let $Q \subset \P^4$ be the smooth quintic. Let $F^{\mathrm{CLW}}_g(Q, t)$
  be the genus-$g$ amplitude produced by Costello--Li--Williams
  renormalised BV quantisation of Kodaira--Spencer gravity, and let
  $F^{\mathrm{GV}}$ be the Gopakumar--Vafa resummation~\eqref{eq:gv-expansion}
  with BPS invariants $n_\beta^g(Q)$ computed by Huang--Klemm--Quackenbush
  2009 through genus $22$. Then in the Kähler cone $\Re t > t_c$ with
  $t_c$ the image of the conifold locus $\psi = 1$ on the complex-structure
  moduli space of the mirror quintic $\widetilde Q$:
  \begin{enumerate}
    \item[(i)] \emph{(Genus-by-genus agreement, verified.)} For every
      $0 \leq g \leq 22$,
      \[
        F^{\mathrm{CLW}}_g(Q, t) \;=\; [\lambda^{2g - 2}]\, F^{\mathrm{GV}}(\lambda, t)
      \]
      as formal power series in $e^{-\beta \cdot t}$; equality holds in
      $\Z[[e^{-t}]]$ after the Katz--Klemm--Vafa integrality normalisation.
    \item[(ii)] \emph{(Convergence of~\eqref{eq:gv-expansion} at fixed genus.)}
      The genus-$0$ BPS series $\sum_\beta n_\beta^0 e^{-\beta t}$
      converges absolutely on $\Re t > t_c$, with $t_c$ finite and equal
      to the entropy temperature of genus-$0$ BPS states; the
      Castelnuovo bound $n_\beta^0 \leq \binom{\beta \cdot H + 2}{2} + O(\beta)$
      (HKQ 2009) is the input.
    \item[(iii)] \emph{(Stokes match at the conifold.)} The trans-series
      completion of $F^{\mathrm{pert}}$ produced by the GV resummation
      agrees at the conifold point $\psi = 1$ with the Candelas--de la
      Ossa--Green--Parkes 1991 mirror-period monodromy: the Stokes
      constant across the conifold ray is $S_c = 1$, and the instanton
      action is $A = 2\pi i\, \tau_c$ where $\tau_c$ is the
      vanishing-cycle period $\oint_{\gamma_c} \Omega_{\widetilde Q}$.
  \end{enumerate}
\end{thm}

\begin{proof}
  (i) applies the Pandharipande--Thomas formulation of MNOP
  (Theorem~\ref{thm:mnop-quintic}) to the CLW genus-$g$ amplitudes:
  CLW renormalisation of Kodaira--Spencer gravity on $Q$ produces at
  each loop order the BCOV $F_g$ via the holomorphic anomaly recursion,
  with input $F_0$ equal to the CDGP prepotential, which matches the
  PT generating function under $-q = e^{i\lambda}$. HKQ 2009 verify PT
  integrality through $g = 22$ by direct modular bootstrap of the
  propagator ring combined with Castelnuovo-type bounds on $n_\beta^g$;
  their computer-assisted computation is primary source.

  (ii) The Castelnuovo bound gives $\log n_\beta^0 = O(\log \beta)$;
  combined with the genus-zero MacMahon-type generating function,
  $\sum_\beta n_\beta^0 e^{-\beta t}$ converges for $\Re t > 0$, the
  actual radius $t_c$ being fixed by the conifold period
  $t_c = -\log |\psi_c - 1|$ in the mirror coordinate.

  (iii) Near the conifold,
  $F_g^{\mathrm{pert}}(Q, t) \sim (2g - 3)! / (2\pi i\, \tau_c)^{2g - 2}$
  (Huang--Klemm 2007), which is exactly the leading-$m$ contribution to
  \eqref{eq:gv-expansion} at $\beta = \beta_c$ (the conifold-curve class)
  with $n_{\beta_c}^0 = 1$. The factorial growth of the perturbative
  series is matched by the residues at $\lambda = 2\pi / m$ of the GV
  resummation; the Écalle--Borel resurgent sum is unique once the
  Stokes constant $S_c = 1$ is fixed by the monodromy of the mirror
  period across the conifold ray. This is the Cecotti--Codesido--Marino
   2013 resurgent-completion protocol, implemented at the quintic with
  HKQ data as input.\qed
\end{proof}

\begin{rmk}[What (iii) does \emph{not} claim]
  The resurgent completion in (iii) is Borel-summed along the physical
  ray and matched at one Stokes wall (the conifold). A full
  non-perturbative completion requires matching at every Stokes wall of
  the quintic moduli space: large complex structure, conifold, and
  Landau--Ginzburg / Gepner. At Gepner, HKQ 2009 fix $F_g$ through
  genus $22$ by modular bootstrap; at large complex structure, the GV
  expansion~\eqref{eq:gv-expansion} is the definition. The residual
  obstruction is the \emph{global} resurgent sheaf: a unique trans-series
  $F^{\mathrm{np}}(\lambda, t)$ on the full moduli space whose Stokes
  jumps match the BCOV monodromy representation on $H^3(\widetilde Q, \Z)$.
  This is open beyond the quintic and open globally on the quintic
  moduli space; it is closed locally at each of the three special loci.
\end{rmk}

\begin{conj}[Mirror-period recovery from non-perturbative BCOV]
  \label{conj:mirror-period-recovery}
  Let $\varpi_0(\psi), \varpi_1(\psi), \varpi_2(\psi), \varpi_3(\psi)$
  be the CDGP 1991 fundamental periods of the mirror quintic
  $\widetilde Q$, solutions of the quintic Picard--Fuchs operator
  \[
    \mathcal{L}_{\mathrm{PF}} \;=\; (\psi \partial_\psi)^4
      - 5\psi\, \prod_{k = 1}^{4} (5 \psi \partial_\psi + k).
  \]
  The genus-zero prepotential extracted from
  $F^{\mathrm{GV}}(\lambda, t) = \lambda^{-2}\, \mathcal{F}_0(t) + O(\lambda^0)$
  with $t = \varpi_1 / \varpi_0$ the mirror map satisfies
  \[
    \mathcal{F}_0(t) \;=\; \tfrac{1}{\varpi_0^2}\,
      \bigl(\varpi_0 \varpi_3 - \varpi_1 \varpi_2\bigr)
      + \text{(classical polynomial)}
  \]
  exactly, with no instanton corrections beyond those encoded in
  $\{n_\beta^0\}_\beta$. Small-$\beta$ check: $n_1^0 = 2875$,
  $n_2^0 = 609250$, $n_3^0 = 317206375$ (CDGP 1991; Schubert 1886;
  Katz 1986; Ellingsrud--Strømme 1995) reproduce the CDGP $q$-expansion
  of the normalised Yukawa coupling to all orders tested.
\end{conj}

\begin{rmk}[Relation to the chiral-volume conjecture]
  Conjecture~\ref{conj:mirror-period-recovery} is the genus-zero shadow

  HKQ genus-$22$ data closes this link through genus $22$; higher
  genera require the resurgent completion of
  Theorem~\ref{thm:bcov-np-quintic}(iii).
\end{rmk}
