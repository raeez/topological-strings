\section{Acceptance criteria and obstruction complexes}\label{sec:open-obligations}

The universal formal-local Koszul duality programme is organized by
theorem data.  Each item below is either a construction already used in
the local theorem, an acceptance criterion for a frontier extension, or
an obstruction complex whose vanishing is the exact condition for the
next theorem.

\begin{enumerate}
  \item \emph{Finite cotangent CE/PV data.}
  A finite window consists of
  \(\mathfrak h_{\le M}\), a finite dual \(D_{\le M}\), and the
  cotangent Lie algebra
  \[
    \mathfrak g_{\le M}=\mathfrak h_{\le M}\ltimes D_{\le M}[1].
  \]
  The finite comparison is
  \[
    \Phi_M\colon
    C^\bullet_{\mathrm{CE}}(\mathfrak g_{\le M})
    \xrightarrow{\sim}
    \mathrm{PV}(\mathbf S(\mathfrak h_{\le M})),
    \qquad c\mapsto\theta,\quad u\mapsto O .
  \]

  \item \emph{Completed formal-disk CE/PV acceptance.}
  The inverse-limit comparison for
  \(\mathfrak h=\C[[z_1,z_2]]/\C\cdot1\) and
  \(\mathfrak g=\mathfrak h\ltimes
  \mathfrak h^\vee_{\mathrm{cont}}[1]\) is accepted when the finite
  windows form a CE/PV-admissible inverse system.  The obstruction
  terms are
  \[
    \mathfrak o^{\mathrm{br}}_{M,N}(x,y)
      =\pi_{M,N}[x,y]_M-[\pi_{M,N}x,\pi_{M,N}y]_N,
  \]
  \[
    \mathfrak o^{\mathrm{coad}}_{M,N}(x,\lambda)
      =\pi^\vee_{M,N}(x\cdot_M\lambda)
       -(\pi_{M,N}x)\cdot_N(\pi^\vee_{M,N}\lambda),
  \]
  and
  \[
    \mathfrak o^{\mathrm{Cas}}_{M,N}
      =(\pi_{M,N}\otimes\pi^\vee_{M,N})C_M-C_N .
  \]
  Their vanishing gives
  \[
    C^\bullet_{\mathrm{CE},\mathrm{cont}}(\mathfrak g)
    \xrightarrow{\sim}
    \mathrm{PV}(\widehat{\mathbf S}_{\mathrm{Tate}}(\mathfrak h)).
  \]

  \item \emph{Admissible Tate bar-cobar/Quillen envelope.}
  For an admissible Tate replacement
  \(\mathfrak g_{\mathrm{adm}}\), put
  \[
    B_{\mathrm{adm}}=
    C^\bullet_{\mathrm{CE},\mathrm{cont}}(\mathfrak g_{\mathrm{adm}}).
  \]
  The Koszul duality theorem in the envelope is
  \[
    B_{\mathrm{adm}}^!\simeq
    \widehat U(\mathfrak g_{\mathrm{adm}}),
  \]
  together with the bar-cobar Quillen equivalence under conilpotency,
  PBW completeness, and the completed symmetric associated graded.

  \item \emph{Closed-open generator map.}
  The closed-open map is determined on CE/PV generators by
  \[
    c^I\mapsto\theta^I,\qquad u_I\mapsto O_I .
  \]
  The boundary Hamiltonian observable is sourced by the shifted
  cotangent coordinate \(u_I\).  This type check is part of the theorem:
  \(c^I\) gives the vector-field operation, while \(u_I\) gives the
  Hamiltonian boundary coordinate.

  \item \emph{Matlis/local-cohomology cotangent source.}
  The continuous cotangent module is
  \[
    \mathcal P=
    \operatorname{Ann}_{\mathrm{res}}(\C\cdot1)
    \subset H^2_{\mathfrak m}(\C[[z_1,z_2]])\,dz_1dz_2,
  \]
  with basis
  \[
    \rho_{a,b}=z_1^{-a-1}z_2^{-b-1}dz_1dz_2,\qquad a+b>0.
  \]
  The coadjoint action is
  \[
    z_1^p z_2^q\cdot\rho_{a,b}
    =
    -(pb-qa+p-q)\rho_{a-p+1,b-q+1}.
  \]

  \item \emph{Primitive higher-\(\psi\) Hamiltonian extraction.}
  The full primitive Koszul complex for \(Q\psi=[x,y]\) extracts the
  Hamiltonian sectors:
  nonconstant \(H_0\) classes, primitive one-\(\psi\) classes
  \(\Psi_{a,b}\), and no primitive \(p\ge2\) homology.  The one-\(\psi\)
  action is
  \[
    \{\overline{\operatorname{Tr}(\phi_1^p\phi_2^q)},
      \widetilde\Psi_{a,b}\}
    =(pb-qa)\widetilde\Psi_{a+p-1,b+q-1},
  \]
  with negative targets zero and scalar targets killed in the
  Hamiltonian quotient.

  \item \emph{Principal-part versus polynomial-descendant criterion.}
  The polynomial one-\(\psi\) and principal-part modules have opposite
  linear-Hamiltonian behavior:
  \[
    \{O_{1,0},\widetilde\Psi_{a,b}\}
      =b\,\widetilde\Psi_{a,b-1},
    \qquad
    \{O_{0,1},\widetilde\Psi_{a,b}\}
      =-a\,\widetilde\Psi_{a-1,b},
  \]
  while
  \[
    z_1\cdot\rho_{a,b}=-(b+1)\rho_{a,b+1},
    \qquad
    z_2\cdot\rho_{a,b}=(a+1)\rho_{a+1,b}.
  \]
  This local-nilpotence obstruction is the acceptance criterion for any
  proposed polynomial realization of \(\mathcal P\): it must change the
  target, the action, or the category.

  \item \emph{\(\mathcal D_\hbar\) Fourier--Rees bridge.}
  With
  \[
    \mathcal D_\hbar
    =
    \C[[\hbar]]
      \langle z_1,z_2,\partial_1,\partial_2\rangle/
      ([\partial_i,z_j]=\hbar\delta_{ij}),
  \]
  the Fourier delta module
  \(\Delta^{\mathsf F}_\hbar=\mathcal D_\hbar/(z_1,z_2)\) gives
  \[
    z_1^az_2^b\mapsto
    \partial_1^a\partial_2^be_0
    \mapsto
    (-1)^{a+b}\hbar^{a+b}a!b!\rho_{a,b}.
  \]
  Before \(\hbar\) is inverted, the image is the Rees sublattice
  \(\bigoplus\hbar^{a+b}\C[[\hbar]]\rho_{a,b}\).  The same-action
  obstruction forbids a flat Rees family that identifies polynomial
  one-\(\psi\) descendants with \(\mathcal P\) as modules for the
  original Hamiltonian action.

  \item \emph{One-antifield Moyal target.}
  The reduced distributional principal-part target has bracket
  \[
    \{B_f(a),\Theta_\rho(B)\}_\hbar
      =
      \Theta_{\operatorname{ad}^{\vee}_{f,\hbar}\rho}(aB),
    \qquad
    \{\Theta_\rho(B),\Theta_\sigma(C)\}_\hbar=0.
  \]
  A strict quantum descendant lift of the classical label shadow is
  excluded by the leading-term calculation
  \[
    z_1\cdot_0\widetilde\Psi_{1,0}=0,\qquad
    z_1\cdot\rho_{1,0}=-\rho_{1,1}.
  \]

  \item \emph{Weighted finite-window regulator independence.}
  A regulator-admissible sector is a strict Mittag-Leffler family on
  finite windows with graphwise bounded Hamiltonian and cotangent legs.
  For admissible weights \(w,w'\), diagonal transports
  \(R^{w_0}_{w,w'}\) identify the local-functional complexes and carry
  the obstruction class
  \[
    \mathrm{Ob}_w\in
    H^1_{\mathrm{adm}}(\mathcal O_{\mathrm{loc}}(\mathcal E_w),
      Q+\{I_0^w,-\})
  \]
  to \(\mathrm{Ob}_{w'}\).  Thus QME vanishing is invariant under
  admissible finite-window weight.

  \item \emph{Endpoint admissibility.}
  An endpoint for \(w_q(d)=q^d\) is accepted by conditions (E1)--(E5):
  strict Mittag-Leffler quotients, a coefficient kernel projecting to
  every finite Casimir, transport of differential, bracket, coadjoint
  action, propagator, interaction, counterterms, and the generator map
  \(c\mapsto\theta,\ u\mapsto O\), admissible filtered cohomology, and
  RG preservation.  The endpoint obstruction tuple is
  \[
    \mathfrak o_{\mathrm{BV,end}}
    =
    (\mathfrak o_{\mathrm{Cas}},
     \mathfrak o_{\mathrm{def}},
     \mathfrak o_{\mathrm{RG}},
     \mathfrak o_{\mathrm{QME}}).
  \]
  The strict product/direct-sum pair fails the criterion because no
  tensor \(K\) projects to all finite Casimirs
  \(C^M_{\mathfrak h}\).

  \item \emph{Mixed brane-defect QME complex.}
  The local-functional obstruction lives in
  \[
    H^1\bigl(\mathcal O_{\mathrm{loc}}(\mathcal E_w),
      Q+\{I_0,-\}\bigr).
  \]
  Its scalar component is computed by
  \[
    \{I_\partial(a,f),I_\partial(b,g)\}_{\mathrm{open}}
    =
    I_\partial(ab,\{f,g\}_{\bar A})
      +N\,\omega(f,g)\int_\R a(t)b(t)\,dt ,
  \]
  or, in Weyl normalization,
  \[
    \operatorname{Tr}(A\,XY)-\operatorname{Tr}(A\,YX)
      =\hbar N\,\operatorname{Tr}(A)
      \quad\bmod\ \mathcal W_N\widehat\mu(\mathfrak{gl}_N).
  \]
  With the central ghost \(\gamma_{\mathbf 1}\), the representative is
  \[
    \operatorname{Ob}_{\mathrm{sc}}
    =
    \hbar N\,\omega(f,g)
    \int_\R a(t)b(t)\gamma_{\mathbf 1}(t)\,dt .
  \]
  A scalar-reduced ordinary \(\mathfrak{gl}_N\) lift requires
  \(W_{\mathrm{ext}}\) with leading class
  \([w_{\mathrm{ext}}]=-\hbar N[\bar c]\), or a central-character /
  balanced-supertrace mechanism that kills the same line.

  \item \emph{Unreduced factorization-current lift.}
  The unreduced cotangent lift is a datum:
  distributional principal-part boundary fields, product centrality
  homotopies, \(P_0\)-bracket homotopies, a coefficient kernel, a
  brane-defect propagator, and a QME solution.  Polynomial boundary
  representatives are ruled out by the local-nilpotence theorem; the
  accepted target is the principal-part current sector with the
  centrality homotopies displayed as part of the theorem data.

  \item \emph{Moyal and graph normalization.}
  The Weyl/Moyal convention is
  \[
    f*g=m\circ\exp\left(\frac{\hbar}{2}
    (\partial_{z_1}\otimes\partial_{z_2}
     -\partial_{z_2}\otimes\partial_{z_1})\right)(f\otimes g).
  \]
  The normalized bracket is
  \[
    \{f,g\}_\hbar
    =\sum_{s\ge0}
      \frac{\hbar^{2s}}{2^{2s}(2s+1)!}P^{2s+1}(f,g).
  \]
  The first and third raw commutator weights are \(1\) and \(1/24\);
  any Costello realization must reproduce these coefficients after its
  sign, orientation, automorphism, and counterterm conventions are fixed.

  \item \emph{Global and cross-volume descent.}
  A target theorem supplies a matched-conventions datum
  \[
  \mathfrak C_{\mathrm{tar}}=
  (d,Y,\eta,(\Sigma_{d-1},C),L,\tau,\lambda,\mathfrak f,\Lambda,
  \hbar_{\mathrm{tar}},\mathcal O_{\mathrm{tar}},
  \chi_{\mathrm{cl}},\chi_{\mathrm{op}},\chi_{\mathrm{anom}},
  \mathcal K_{\mathrm{obs}}).
  \]
  Its acceptance vector is
  \[
    \operatorname{Ob}_{\mathfrak C_{\mathrm{tar}}}
    =
    (\operatorname{per}_{\mathrm{tar}},
     \operatorname{ob}_{\mathrm{WR}},
     \operatorname{ob}_{\mathrm{anom}},
     \operatorname{ob}_{\mathrm{QME}},
     \operatorname{ob}_{\mathrm{MD}}).
  \]
  The formal-local theorem transfers to a compact BCOV, Vol III,
  Igusa/BKM, or sister-volume statement precisely when this vector
  vanishes and the comparison morphisms identify the target formal
  restriction with the local Hamiltonian BF/Moyal normalization.
\end{enumerate}
