\textbf{Formal-local theorem infrastructure.}
The universal formal-local Koszul-duality interface is attached to
the shifted cotangent Hamiltonian Lie algebra
\[
  \mathfrak h=\C[[z_1,z_2]]/\C\cdot1,\qquad
  \mathfrak g=\mathfrak h\ltimes
    \mathfrak h^\vee_{\mathrm{cont}}[1].
\]
Its visible interface is the chain
\[
\begin{gathered}
  \text{finite cotangent CE/PV}\\
  \Downarrow\\
  \text{completed formal-disk CE/PV}\\
  \Downarrow\\
  \text{admissible Tate bar-cobar/Quillen envelope}\\
  \Downarrow\\
  \text{closed-open derived-centre data}.
\end{gathered}
\]
The closed-open generator map is fixed once and for all:
\[
  c^I\longmapsto\theta^I,\qquad u_I\longmapsto O_I .
\]
The theory is extended by the primitive higher-\(\psi\) extraction, the
\(\mathcal D_\hbar\) Fourier--Rees bridge, the weighted finite-window
regulator, the endpoint criteria, and the unreduced factorization-current
lift datum.

\smallskip
\textbf{Acceptance principle.}
Every frontier extension is stated as theorem data plus an obstruction
complex.  No compact BCOV, Vol III,
Igusa/BKM, or global Calabi--Yau statement follows from the formal disk alone;
such statements are accepted exactly by a matched-conventions datum with
vanishing period, anomaly, QME, descent, and operadic obstruction vector.

\paragraph{Theorem data.}

\begin{enumerate}
  \item \emph{Finite cotangent CE/PV.}
  For each finite window,
  \[
    \Phi_M\colon
    C^\bullet_{\mathrm{CE}}
      (\mathfrak h_{\le M}\ltimes D_{\le M}[1])
    \xrightarrow{\sim}
    \mathrm{PV}(\mathbf S(\mathfrak h_{\le M}))
  \]
  with \(c\mapsto\theta\), \(u\mapsto O\).  The coordinate comparison
  assembles as a pro finite-window statement and as a completed statement
  only in a mixed or weighted topology where the contraction bracket is
  continuous.  The strict ambient product/direct-sum completion is
  obstructed by the infinite diagonal.

  \item \emph{Completed formal disk CE/PV.}
  The completed formal-disk comparison is a pro finite-window or
  mixed-continuous comparison
  \[
    \Phi\colon
    C^\bullet_{\mathrm{CE},\mathrm{cont}}(\mathfrak g)
    \longrightarrow
    \mathrm{PV}(\widehat{\mathbf S}_{\mathrm{Tate}}(\mathfrak h)).
  \]
  Its coordinate transition obstruction terms are
  \[
    \mathfrak o^{\mathrm{br}}_{M,N},\qquad
    \mathfrak o^{\mathrm{coad}}_{M,N}.
  \]
  The separate strict-kernel and ambient-product bracket obstruction is
  \(\mathfrak o^{\mathrm{Cas}}_{M,N}\).

  \item \emph{Admissible Tate bar-cobar/Quillen envelope.}
  For an admissible Tate replacement
  \(\mathfrak g_{\mathrm{adm}}\),
  \[
    B_{\mathrm{adm}}=
    C^\bullet_{\mathrm{CE},\mathrm{cont}}(\mathfrak g_{\mathrm{adm}}),
    \qquad
    B_{\mathrm{adm}}^!\simeq
    \widehat U(\mathfrak g_{\mathrm{adm}}).
  \]
  The envelope hypotheses are conilpotent CE coalgebra, PBW
  completeness, completed symmetric associated graded, and an admissible
  transferred \(P_0\)-algebra model structure.

  \item \emph{Closed-open generator map.}
  The Hamiltonian CE ghost \(c^I\) becomes the Schouten vector
  \(\theta^I\).  The shifted-cotangent CE coordinate \(u_I\) becomes the
  Hamiltonian boundary coordinate \(O_I\).  Boundary Hamiltonian
  observables come from \(u_I\).

  \item \emph{Dirac open Koszul model.}
  The brane stack is the derived commuting-pair locus:
  \[
    S_\partial=\int_\R\operatorname{Tr}
    (\phi_1\,d\phi_2+A[\phi_1,\phi_2]),\qquad
    Q\psi=[\phi_1,\phi_2].
  \]
  Stable trace invariants are read degreewise; for total word degree
  \(d\), the stable range is \(N\ge d\).

  \item \emph{Primitive higher-\(\psi\) extraction.}
  The primitive Koszul complex has nonconstant Hamiltonian \(H_0\)
  classes and positive Hamiltonian one-\(\psi\) primitive \(H_1\)
  classes \(\Psi_{a,b}\) with \(a+b>0\).  The scalar
  \(\Psi_{0,0}=\operatorname{Tr}(\psi)\) is the removed odd scalar line,
  not a Matlis principal-part class.  The higher primitive sectors
  satisfy \(H_p=0\) for \(p\ge2\) in the stable finite-bidegree cyclic
  word model.  The
  one-\(\psi\) action is
  \[
    \{\overline{\operatorname{Tr}(\phi_1^p\phi_2^q)},
      \widetilde\Psi_{a,b}\}
    =(pb-qa)\widetilde\Psi_{a+p-1,b+q-1}.
  \]

  \item \emph{Matlis cotangent module.}
  The cotangent coefficient module is
  \[
    \mathcal P=
    \operatorname{Ann}_{\mathrm{res}}(\C\cdot1)
    \subset H^2_{\mathfrak m}(R)\,dz_1dz_2,
    \qquad
    \rho_{a,b}=z_1^{-a-1}z_2^{-b-1}dz_1dz_2 .
  \]
  Its coadjoint action is
  \[
    z_1^pz_2^q\cdot\rho_{a,b}
    =-(pb-qa+p-q)\rho_{a-p+1,b-q+1}.
  \]

  \item \emph{Polynomial-descendant obstruction.}
  Polynomial one-\(\psi\) representatives have locally nilpotent linear
  Hamiltonian action, while
  \[
    z_1\cdot\rho_{a,b}=-(b+1)\rho_{a,b+1},\qquad
    z_2\cdot\rho_{a,b}=(a+1)\rho_{a+1,b}
  \]
  is not locally nilpotent on principal parts.  Any polynomial
  realization of \(\mathcal P\) must therefore change the target, the
  action, or the category.

  \item \emph{\(\mathcal D_\hbar\) Fourier--Rees bridge.}
  With
  \[
    \mathcal D_\hbar=
    \C[[\hbar]]\langle z_i,\partial_i\rangle/
    ([\partial_i,z_j]=\hbar\delta_{ij}),
  \]
  the Fourier delta lattice gives
  \[
    z_1^az_2^b\mapsto\partial_1^a\partial_2^be_0
    \mapsto(-1)^{a+b}\hbar^{a+b}a!b!\rho_{a,b}.
  \]
  Before inverting \(\hbar\) the image is the Rees sublattice
  \(\bigoplus\hbar^{a+b}\C[[\hbar]]\rho_{a,b}\).  The bridge changes
  polarization; same-action Rees families from \(\Psi_{a,b}\) to
  \(\rho_{a,b}\) are obstructed.

  \item \emph{Quantum Hamiltonian and Moyal sector.}
  The Weyl convention is
  \[
    f*g=m\exp\!\left(\frac{\hbar}{2}P\right)(f\otimes g),
    \qquad
    \{f,g\}_\hbar=\hbar^{-1}[f,g]_* .
  \]
  The first and third raw commutator coefficients are \(1\) and
  \(1/24\).  The descendant sector is governed by the principal-part
  Moyal target and the no-\(\hbar\)-adic same-action intertwiner.

  \item \emph{Weighted finite-window regulator.}
  Admissible weights \(w,w'\) are related by diagonal finite-window
  transport \(R_{w,w'}^{w_0}\).  The regulator theorem identifies the
  Mittag-Leffler local-functional complexes and carries
  \(\mathrm{Ob}_w\) to \(\mathrm{Ob}_{w'}\).  Vanishing of the mixed
  brane-defect QME class is independent of the admissible weight.

  \item \emph{Endpoint criteria.}
  An endpoint for \(w_q(d)=q^d\) is admissible when conditions
  (E1)--(E5) hold.  The obstruction tuple is
  \[
    \mathfrak o_{\mathrm{BV,end}}=
    (\mathfrak o_{\mathrm{Cas}},
     \mathfrak o_{\mathrm{def}},
     \mathfrak o_{\mathrm{RG}},
     \mathfrak o_{\mathrm{QME}}).
  \]
  The strict product/direct-sum endpoint fails because no tensor
  projects to all finite-window Casimirs \(C_{\mathfrak h}^{w_0}\).

  \item \emph{Unreduced factorization-current lift.}
  Distributional principal-part currents give the target for cotangent
  boundary modes.  The lift datum consists of boundary representatives,
  product centrality homotopies, \(P_0\)-bracket homotopies, the
  coefficient kernel, the brane-defect propagator, and the QME solution.
  The scalar component is
  \[
    \operatorname{Ob}_{\mathrm{sc}}
    =\hbar N\,\omega(f,g)
      \int_\R a(t)b(t)\gamma_{\mathbf 1}(t)\,dt ,
  \]
  with associated graded \(\hbar N[\bar c]\).  Central-character,
  balanced-supertrace, or closed-exchange data must kill this class.

  \item \emph{Global and cross-volume descent.}
  A target theorem supplies a matched-conventions datum
  \(\mathfrak C_{\mathrm{tar}}\).  The acceptance vector is
  \[
    \operatorname{Ob}_{\mathfrak C_{\mathrm{tar}}}
    =
    (\operatorname{per}_{\mathrm{tar}},
     \operatorname{ob}_{\mathrm{WR}},
     \operatorname{ob}_{\mathrm{anom}},
     \operatorname{ob}_{\mathrm{QME}},
     \operatorname{ob}_{\mathrm{MD}}).
  \]
  It must vanish, with compatible null-homotopies, before any compact
  BCOV, Vol III, Igusa/BKM, or global Calabi--Yau consequence is read
  from the formal-local theorem.
\end{enumerate}
